CREATE DATABASE eAsset;

create table credential(userName varchar(32) primary Key,password varchar(32));

create table category(AssetType varchar(32) primary key,lendingPeriod int,lateFee float(9,2),bannedDaysCount int);

create table user(id int primary key auto_increment,name varchar(32),role varchar(16),phone varchar(10),email varchar(64),userName varchar(32),constraint fk_userName foreign key (userName) references credential(userName));

create table asset(id int primary key auto_increment,name varchar(32),assetType varchar(32),description varchar(32),dateOfAddition date,isAvailable boolean,constraint fk_assetType foreign key (assetType) references category(assetType));
